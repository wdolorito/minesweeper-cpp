configure:
CC=clang CXX=clang++ ../wxWidgets-3.0.3/configure --enable-stl --with-macosx-version-min=10.10 --prefix=$HOME/local
make BUILD=debug

build project:
add ~/local/bin to .profile path (PATH=~/local/bin:$PATH)
make {release}

run project:
cd packages/wxMac
./package_mac.sh
open Minesweep.app
